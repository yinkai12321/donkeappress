# 列配置 columns.js

常用配置

配置项 默认值 说明
field 列字段名
title 列标题
width 列宽度
visible true 默认是否显示
resizable true 列是否允许拖动列宽调整大小
align 列对齐方式
sortable false 是否允许列排序
tree-node false 只对 tree-config 配置时有效，指定为树节点
formatter 格式化显示内容

其他配置详见 https://xuliangzhan_admin.gitee.io/vxe-table/#/column/api (opens new window)

# 列配置示例

switch

{
    field: "enabledMark",
    title: "是否显示菜单",
    width: 100,
    slots: {
        // 使用 JSX 渲染
        default: ({ row }) => {
            return [
                <vxe-switch
                    open-value={1}
                    close-value={0}
                    v-model={row.enabledMark}
                ></vxe-switch>,
            ];
        },
    },
}

# 表单配置 fields.js

配置项 类型 说明 示例
field string 字段名
title string 标题
type string 表单输入类型：input、inputnumber、select、selectTree、date、switch、checkbox、radio、textarea、dialogselect "input"
colwidth int 表单占用列数 值为 1、2、3、4 1
labelCol Object label 标签布局，同 <Col> 组件，设置 span offset 值，如 {span: 3, offset: 12} 或 sm: {span: 3, offset: 12} { style: "width: 120px" }
wrapperCol Object 需要为输入控件设置布局样式时，使用该属性，用法同 labelCol { style: "width: calc(100% - 120px);" }
url string 下拉框数据项请求地址 `${CONFIG.base.ip}/api/Base_Menu/GetListAll`
param Object 下拉框数据项请求参数 {name:"123"}
result string 下拉框数据项返回结果参数 "data"
datavalue string 下拉框数据项编号字段 "EquiNo"
datalabel string 下拉框数据项名称字段 "EquiName"
bindData string 下拉框数据项绑定名称字段 "EquiName"
selectData array 下拉框数据项列表 [{ key: "0",title:"顶级节点",}]
multiple bool 树状下拉框是否多选(只用于树状下拉框) false
mode string 普通下拉框选择的模式:default 单选、 multiple 多选、tags 可自定义标签、combobox 可输入(只用于普通下拉框) default
reqFunction Function 弹出框请求方法 getListAll
selectType string 弹出框选择类型：单选radio 多选 checkbox checkbox
selectValue string 弹出框选择字段 ManNo
title string 弹出框标题 选择员工
selectColumns Array 弹出框列配置 配置示例如下

selectColumns 配置示例

[
    {
        title: "工号",
        dataIndex: "ManNo",
        key: "ManNo",
    },
    {
        title: "人员姓名",
        dataIndex: "ManName",
        key: "ManName",
    },
    {
        title: "性别",
        dataIndex: "Sex",
        key: "Sex",
    },
],

# 表单配置示例

机构分类（下拉框）

{
    label: "机构分类",
    type: "select",
    field: "Category",
    // eslint-disable-next-line no-undef
    selectData: [
        {
            key: "0",
            title: "事业单位",
            value: "0",
        },
        {
            key: "1",
            title: "国企",
            value: "1",
        }
    ],
    rule: [
        { required: true, message: '请输入机构分类', trigger: 'blur' },
    ],
},

上级（树状下拉框）

{
    label: "上级",
    type: "selectTree",
    field: "parentId",
    // eslint-disable-next-line no-undef
    url: `${CONFIG.base.ip}/api/Permission/Organize/Selector`,
    datavalue: "Id",
    datalabel: "fullName",
    selectData: [{
        key: "0",
        title: "顶级节点",
        value: "0",
    }],
    rule: [
        { required: true, message: '请输入父级主键', trigger: 'blur' },
    ],
},

岗位（树状下拉框多选）

{
    label: "岗位",
    type: "selectTree",
    field: "PositionId",
    rule: [
        { required: true, message: '请输入岗位', trigger: 'blur' },
    ],
    // eslint-disable-next-line no-undef
    url: `${CONFIG.base.ip}/api/Permission/Position/Selector`,
    datavalue: "Id",
    datalabel: "fullName",
    selectData: [],
    multiple: true,//是否多选
},

角色（下拉框多选）

{
    label: "角色",
    type: "select",
    field: "RoleId",
    rule: [
        { required: true, message: '请输入角色', trigger: 'blur' },
    ],
    // eslint-disable-next-line no-undef
    url: `${CONFIG.base.ip}/api/Permission/Role/Selector`,
    datavalue: "id",
    datalabel: "fullName",
    selectData: [],
    mode: "multiple",//多选
},

员工工号（弹出框）

{
    label: "员工工号",
    type: "dialogselect",
    field: "ManId",
    rule: [
        { required: true, message: '员工工号', trigger: 'change' },
    ],
    reqFunction: getListAll,//请求方法
    selectType: "checkbox", // 单选radio  多选 checkbox
    selectValue: "ManNo",   //选择字段
    title: "选择员工",
    selectColumns: [
        {
            title: "工号",
            dataIndex: "ManNo",
            key: "ManNo",
        },
        {
            title: "人员姓名",
            dataIndex: "ManName",
            key: "ManName",
        },
        {
            title: "性别",
            dataIndex: "Sex",
            key: "Sex",
        },
    ],
},
